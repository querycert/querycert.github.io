Definition string_to_enhanced_type (s:string) : option enhanced_type :=
  match s with
  | "ETop"%string => Some enhancedTop
  | "EBottom"%string => Some enhancedBottom
  | "ESqlDate"%string => Some enhancedSqlDate

  | _ => None
  end.
