  Definition pd_tbindings := list (string*option rtype).

  Definition preserves_Some {A B} (f:A->option B) (xd yd:A) : Prop
    := forall xd', f xd = Some xd' -> exists yd', f yd= Some yd'.

  Global Instance preserves_Some_pre {A B} (f:A->option B)
    : PreOrder (preserves_Some f).
Proof.
    constructor; red; unfold preserves_Some; firstorder.
  Qed.


  Lemma Forall2_preserves_Some_snd_update_first {B} l v (d:B) :
    Forall2 (preserves_Some snd) l (update_first string_dec l v (Some d)).
Proof.
    induction l; simpl; trivial.
    destruct a.
    match_destr.
    - subst.
      constructor; try reflexivity.
      red; simpl; intros; subst; eauto.
    - constructor; eauto.
      reflexivity.
  Qed.
  
  Definition pd_bindings_type (b:pd_bindings) (t:pd_tbindings)
    := Forall2 (fun xy1 xy2 =>
                  (fst xy1) = (fst xy2)
                  /\ lift2Pr data_type (snd xy1) (snd xy2)) b t.

  Definition mc_bindings_type (b:mc_bindings) (t:mc_tbindings)
    := Forall2 (fun xy1 xy2 =>
                  (fst xy1) = (fst xy2)
                  /\ Forall (fun d => data_type d (snd xy2)) (snd xy1)) b t.

  Definition md_bindings_type (b:md_bindings) (t:md_tbindings)
    := Forall2 (fun xy1 xy2 =>
                  (fst xy1) = (fst xy2)
                  /\ forall d, Some d = snd xy1 ->
                               data_type d (snd xy2)) b t.

    Inductive nnrc_impish_expr_type : pd_tbindings -> nnrc_impish_expr -> rtype -> Prop :=
    | type_NNRCimpishGetConstant {τ} Γ s :
        tdot Γc s = Some τ ->
        [ Γ ⊢ NNRCimpishGetConstant s ▷ τ ]
    | type_NNRCimpishVar {τ} Γ v :
        lookup equiv_dec Γ v = Some (Some τ) ->
        [ Γ ⊢ NNRCimpishVar v ▷ τ ]
    | type_NNRCimpishConst {τ} Γ c :
        normalize_data brand_relation_brands c ▹ τ ->
        [ Γ ⊢ NNRCimpishConst c ▷ τ ]
    | type_NNRCimpishBinop {τ₁ τ₂ τ} Γ b e₁ e₂ :
        binary_op_type b τ₁ τ₂ τ ->
        [ Γ ⊢ e₁ ▷ τ₁ ] ->
        [ Γ ⊢ e₂ ▷ τ₂ ] ->
        [ Γ ⊢ NNRCimpishBinop b e₁ e₂ ▷ τ ]
    | type_NNRCimpishUnop {τ₁ τ} Γ u e :
        unary_op_type u τ₁ τ ->
        [ Γ ⊢ e ▷ τ₁ ] ->
        [ Γ ⊢ NNRCimpishUnop u e ▷ τ ]
    | type_NNRCimpishGroupBy {τl k pf} Γ g sl e :
        sublist sl (domain τl) ->
        [ Γ ⊢ e ▷ Coll (Rec k τl pf) ] ->
        [ Γ ⊢ NNRCimpishGroupBy g sl e ▷ GroupBy_type g sl k τl pf ]
    where
    "[ Γ ⊢ e ▷ τ ]" := (nnrc_impish_expr_type Γ e τ) : nnrc_impish
    .

    Notation "[ Γ ⊢ e ▷ τ ]" := (nnrc_impish_expr_type Γ e τ) : nnrc_impish.


    Fixpoint nnrc_impish_stmt_must_assign (s:nnrc_impish_stmt) (x:var) : Prop
      := match s with
         | NNRCimpishSeq s₁ s₂ =>
           nnrc_impish_stmt_must_assign s₁ x \/ nnrc_impish_stmt_must_assign s₂ x
         | NNRCimpishLet v e₁ s₂ =>
           nnrc_impish_stmt_must_assign s₂ x
         | NNRCimpishLetMut v s₁ s₂ =>
           (v <> x /\ nnrc_impish_stmt_must_assign s₁ x) \/ nnrc_impish_stmt_must_assign s₂ x
         | NNRCimpishLetMutColl v s₁ s₂ =>
           nnrc_impish_stmt_must_assign s₁ x \/ nnrc_impish_stmt_must_assign s₂ x
         | NNRCimpishAssign v e =>
           v = x
         | NNRCimpishPush v e =>
           False
         | NNRCimpishFor v e s₀ =>
           False
         | NNRCimpishIf e s₁ s₂ =>
           nnrc_impish_stmt_must_assign s₁ x /\ nnrc_impish_stmt_must_assign s₂ x
         | NNRCimpishEither e x₁ s₁ x₂ s₂ =>
           nnrc_impish_stmt_must_assign s₁ x /\ nnrc_impish_stmt_must_assign s₂ x
         end.

    Reserved Notation "[ Γ , Δc , Δd ⊢ s ]".

    Inductive nnrc_impish_stmt_type :
      pd_tbindings -> mc_tbindings -> md_tbindings ->
      nnrc_impish_stmt -> Prop :=
    | type_NNRCimpishSeq Γ Δc Δd s₁ s₂ :
        [ Γ , Δc , Δd ⊢ s₁ ] ->
        [ Γ , Δc , Δd ⊢ s₂ ] ->
        [ Γ , Δc , Δd ⊢ NNRCimpishSeq s₁ s₂ ]
    | type_NNRCimpishLet Γ Δc Δd τ x e₁ s₂ :
        [ Γ ⊢ e₁ ▷ τ ] ->
        [ (x,Some τ)::Γ , Δc , Δd ⊢ s₂ ] ->
        [ Γ , Δc , Δd ⊢ NNRCimpishLet x e₁ s₂ ]
    | type_NNRCimpishLetMutDef Γ Δc Δd τ x s₁ s₂ :
        nnrc_impish_stmt_must_assign s₁ x ->
        [ Γ , Δc , (x,τ)::Δd ⊢ s₁ ] ->
        [ (x,Some τ)::Γ, Δc , Δd ⊢ s₂ ] ->
        [ Γ , Δc , Δd ⊢ NNRCimpishLetMut x s₁ s₂ ]
    | type_NNRCimpishLetMutNotUsed Γ Δc Δd x s₁ s₂ :
        [ Γ , Δc , (x,⊤)::Δd ⊢ s₁ ] ->
        [ (x,None)::Γ, Δc , Δd ⊢ s₂ ] ->
        [ Γ , Δc , Δd ⊢ NNRCimpishLetMut x s₁ s₂ ]
    | type_NNRCimpishLetMutColl Γ Δc Δd τ x s₁ s₂ :
        [ Γ , (x,τ)::Δc , Δd ⊢ s₁ ] ->
        [ (x,Some (Coll τ))::Γ, Δc , Δd ⊢ s₂ ] ->
        [ Γ , Δc , Δd ⊢ NNRCimpishLetMutColl x s₁ s₂ ]
    | type_NNRCimpishAssign Γ Δc Δd τ τd x e :
        [ Γ ⊢ e ▷ τ ] ->
        lookup string_dec Δd x = Some τd ->
        τ ≤ τd ->
        [ Γ , Δc , Δd ⊢ NNRCimpishAssign x e ]
    | type_NNRCimpishPush Γ Δc Δd τ τd x e :
        [ Γ ⊢ e ▷ τ ] ->
        lookup string_dec Δc x = Some τd ->
        τ ≤ τd ->
        [ Γ , Δc , Δd ⊢ NNRCimpishPush x e ]
    | type_NNRCimpishFor Γ Δc Δd τ x e₁ s₂ :
        [ Γ ⊢ e₁ ▷ Coll τ ] ->
        [ (x,Some τ)::Γ , Δc , Δd ⊢ s₂ ] ->
        [ Γ , Δc , Δd ⊢ NNRCimpishFor x e₁ s₂ ]
    | type_NNRCimpishIf Γ Δc Δd e s₁ s₂ :
        [ Γ ⊢ e ▷ Bool] ->
        [ Γ , Δc , Δd ⊢ s₁ ] ->
        [ Γ , Δc , Δd ⊢ s₂ ] ->
        [ Γ , Δc , Δd ⊢ NNRCimpishIf e s₁ s₂ ]
    | type_NNRCimpishEither Γ Δc Δd τl τr e x₁ s₁ x₂ s₂ :
        [ Γ ⊢ e ▷ Either τl τr] ->
        [ (x₁,Some τl)::Γ , Δc , Δd ⊢ s₁ ] ->
        [ (x₂,Some τr)::Γ , Δc , Δd ⊢ s₂ ] ->
        [ Γ , Δc , Δd ⊢ NNRCimpishEither e x₁ s₁ x₂ s₂ ]
    where
    "[ Γ , Δc , Δd ⊢ s ]" := (nnrc_impish_stmt_type Γ Δc Δd s) : nnrc_impish
    .

    Notation "[ Γ , Δc , Δd ⊢ s ]" := (nnrc_impish_stmt_type Γ Δc Δd s) : nnrc_impish.
  End typ.

  Notation "[ Γc ; Γ , Δc , Δd ⊢ s ]" := (nnrc_impish_stmt_type Γc Γ Δc Δd s) : nnrc_impish.
