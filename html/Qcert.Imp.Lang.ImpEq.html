        Lemma match_eq_lemma :
      forall A B (l1: option A) (l2:option A) f (z:B),
        l1 = l2 ->
        match l1 with
        | Some x => f x
        | None => z
        end
        =
        match l2 with
        | Some x => f x
        | None => z
        end.

      intuition.
      subst;apply eq_refl.
    Qed.
