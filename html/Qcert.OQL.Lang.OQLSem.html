    Inductive filter_cast_sem: brands -> list data -> list data -> Prop :=
    | filter_cast_sem_nil:
        forall b, filter_cast_sem b nil nil (* ⇒ b ⊢〚[]〛ᶜ ⇓ [] *)
    | filter_cast_sem_cons_success:
        forall b1 b2 d1 dl1 dl2,
          filter_cast_sem b2 dl1 dl2 -> (* b₂ ⊢〚↑dl₁〛ᶜ ⇓ ↑dl₂ *)
          sub_brands h b1 b2 -> (* ∧ b₁ <ᵇ b₂ *)
          filter_cast_sem b2 (* ⇒ b₂ ⊢〚(brand b₁ d₁)::↑dl₁〛ᶜ ⇓ (brand b₁ d₁)::↑dl₂ *)
                          ((dbrand b1 d1)::dl1)
                          ((dbrand b1 d1)::dl2)
    | filter_cast_sem_cons_failure:
        forall b1 b2 d1 dl1 dl2,
          filter_cast_sem b2 dl1 dl2 -> (* b₂ ⊢〚↑dl₁〛ᶜ ⇓ ↑dl₂ *)
          ~(sub_brands h b1 b2) -> (* ∧ ¬ b₁ <ᵇ b₂ *)
          filter_cast_sem b2 ((dbrand b1 d1)::dl1) dl2 (* ⇒ b₂ ⊢〚(brand b₁ d₁)::↑dl₁〛ᶜ ⇓ ↑dl₂ *)
    .

    Lemma filter_cast_correct b l1 l2:
      filter_cast_sem b l1 l2 <->
      filter_cast h b l1 = Some l2.
Proof.
      intros; split; intros.
      - revert l2 H; induction l1; intros.
        + inversion H; subst; intros.
          reflexivity.
        + inversion H; subst.
          unfold filter_cast; simpl.
          * unfold filter_cast in *; simpl.
            case_eq (sub_brands_dec h b1 b); intros; [|congruence]; simpl.
            rewrite (IHl1 dl2); [reflexivity|apply H3].
          * unfold filter_cast in *; simpl.
            case_eq (sub_brands_dec h b1 b); intros; [congruence| ]; simpl.
            rewrite lift_id; apply (IHl1 l2 H3).
      - revert l2 H; induction l1; intros.
        + inversion H.
          unfold filter_cast in *; simpl in *.
          inversion H; econstructor.
        + unfold filter_cast in *; simpl in *.
          destruct a; simpl in *; try congruence.
          case_eq (sub_brands_dec h b0 b); intros; subst;
            rewrite H0 in H.
          * unfold lift in H.
            case_eq
              (lift_flat_map
                 (fun x : data =>
                    match x with
                    | dbrand b' _ => if sub_brands_dec h b' b then Some (x :: nil) else Some nil
                    | _ => None
                    end) l1); intros;
              rewrite H1 in H; simpl in *; try congruence.
            inversion H; subst.
            econstructor.
            apply (IHl1 l H1).
            assumption.
          * unfold lift in H.
            case_eq
              (lift_flat_map
                 (fun x : data =>
                    match x with
                    | dbrand b' _ => if sub_brands_dec h b' b then Some (x :: nil) else Some nil
                    | _ => None
                    end) l1); intros;
              rewrite H1 in H; simpl in *; try congruence.
            inversion H; subst.
            econstructor.
            apply (IHl1 l2 H1).
            assumption.
    Qed.
    
    Inductive oql_expr_sem: oql_expr -> oql_env -> data -> Prop :=
    | sem_OConst : forall d d1 env,
        normalize_data h d = d1 -> (* normalize(d) = d₁ *)
        oql_expr_sem (OConst d) env d1 (* ⇒ Γc ; γ ⊢〚d〛⇓ d₁ *)
    | sem_OVar : forall v d env,
        edot env v = Some d -> (* γ(v) = d *)
        oql_expr_sem (OVar v) env d (* ⇒ Γc ; γ ⊢〚v〛⇓ d₁ *)
    | sem_OTable : forall t d env,
        edot constant_env t = Some d -> (* Γc(T) = d *)
        oql_expr_sem (OTable t) env d (* ⇒ Γc ; γ ⊢〚T〛⇓ d *)
    | sem_OBinop : forall bop e1 e2 d1 d2 d3 env,
        oql_expr_sem e1 env d1 -> (* Γc ; γ ⊢〚e₁〛⇓ d₁ *)
        oql_expr_sem e2 env d2 -> (* Γc ; γ ⊢〚e₂〛⇓ d₂ *)
        binary_op_eval h bop d1 d2 = Some d3 -> (* ∧ d₁ ⊠ d₂ = d₃ *)
        oql_expr_sem (OBinop bop e1 e2) env d3 (* ⇒ Γc ; γ ⊢〚e₁ ⊠ e₂〛⇓ d₃ *)
    | sem_OUnop : forall uop e d1 d2 env,
        oql_expr_sem e env d1 -> (* Γc ; γ ⊢〚e〛⇓ d₁ *)
        unary_op_eval h uop d1 = Some d2 -> (* ∧ ⊞ d₁ = d₂ *)
        oql_expr_sem (OUnop uop e) env d2 (* ⇒ Γc ; γ ⊢〚⊞ e〛⇓ d₂ *)
    | sem_OSFW_true_noorder select_e from_e : forall env tenv d,
        oql_from_sem from_e (env :: nil) tenv -> (* Γc; [γ] ⊢〚↑eᶠ〛ᶠ ⇓ ↑γ₁ *)
        oql_select_sem select_e tenv d -> (* ∧ Γc; ↑γ₁ ⊢〚eˢ〛ˢ ⇓ d *)
        oql_expr_sem (OSFW select_e from_e OTrue ONoOrder) env d (* ⇒ Γc; γ ⊢ SELECT eˢ FROM ↑eᶠ *)
    | sem_OSFW_where_noorder select_e from_e where_e : forall env tenv tenv2 d,
        oql_from_sem from_e (env :: nil) tenv -> (* Γc; [γ] ⊢〚↑eᶠ〛ᶠ ⇓ ↑γ₁ *)
        oql_where_sem where_e tenv tenv2 -> (* ∧ Γc; ↑γ₁ ⊢〚eʷ〛ʷ ⇓ ↑γ₂ *)
        oql_select_sem select_e tenv2 d -> (* ∧ Γc; ↑γ₂ ⊢〚eˢ〛ˢ ⇓ d *)
        oql_expr_sem (OSFW select_e from_e (OWhere where_e) ONoOrder) env d (* ⇒ Γc; γ ⊢ SELECT eˢ FROM ↑eᶠ WHERE eʷ *)
    | sem_OSFW_true_order select_e from_e order_e sc : forall env tenv tenv1 d,
        oql_from_sem from_e (env :: nil) tenv -> (* Γc; [γ] ⊢〚↑eᶠ〛ᶠ ⇓ ↑γ₁ *)
        oql_order_sem order_e sc tenv tenv1 -> (* ∧ Γc; ↑γ₁ ⊢〚eᵒ〛ᵒ ⇓ ↑γ₂ *)
        oql_select_sem select_e tenv1 d -> (* ∧ Γc; ↑γ₂ ⊢〚eˢ〛ˢ ⇓ d *)
        oql_expr_sem (OSFW select_e from_e OTrue (OOrderBy order_e sc)) env d (* ⇒ Γc; γ ⊢ SELECT eˢ FROM ↑eᶠ ORDER BY e *)
    | sem_OSFW_where_order select_e from_e where_e order_e sc : forall env tenv tenv1 tenv2 d,
        oql_from_sem from_e (env :: nil) tenv -> (* Γc; [γ] ⊢〚↑eᶠ〛ᶠ ⇓ ↑γ₁ *)
        oql_where_sem where_e tenv tenv1 -> (* ∧ Γc; ↑γ₁ ⊢〚eʷ〛ʷ ⇓ ↑γ₂ *)
        oql_order_sem order_e sc tenv1 tenv2 -> (* ∧ Γc; ↑γ₂ ⊢〚eᵒ〛ᵒ ⇓ ↑γ₃ *)
        oql_select_sem select_e tenv2 d -> (* ∧ Γc; ↑γ₃ ⊢〚eˢ〛ˢ ⇓ d *)
        oql_expr_sem (OSFW select_e from_e (OWhere where_e) (* ⇒ Γc; γ ⊢ SELECT eˢ FROM ↑eᶠ WHERE eʷ ORDER BY eᵒ *)
                           (OOrderBy order_e sc)) env d
    with oql_from_sem: list oql_in_expr -> list oql_env -> list oql_env -> Prop :=
    | sem_OFrom_Nil : forall tenv,
        oql_from_sem nil tenv tenv (* Γc; [γ] ⊢〚[]〛ᶠ ⇓ [] *)
    | sem_OFrom_In in_v e from_e : forall tenv1 tenv2 tenv3,
        oql_from_in_sem in_v e tenv1 tenv2 -> (* Γc; ↑γ₁; v ⊢〚e₀〛ⁱ ⇓ ↑γ₂ *)
        oql_from_sem from_e tenv2 tenv3 -> (* ∧ Γc; ↑γ₂ ⊢〚↑eᶠ〛ᶠ ⇓ ↑γ₃ *)
        oql_from_sem ((OIn in_v e)::from_e) tenv1 tenv3 (* ⇒ Γc; ↑γ₁ ⊢〚(v IN e₀)::↑eᶠ〛ᶠ ⇓ ↑γ₃ *)
    | sem_OFrom_InCast in_v brands e from_e : forall tenv1 tenv2 tenv3,
        oql_from_in_cast_sem in_v brands e tenv1 tenv2 -> (* Γc; ↑γ₁; v; b ⊢〚e₀〛ᶜ ⇓ ↑γ₂ *)
        oql_from_sem from_e tenv2 tenv3 -> (* ∧ Γc; ↑γ₂ ⊢〚↑eᶠ〛ᶠ ⇓ ↑γ₃ *)
        oql_from_sem ((OInCast in_v brands e)::from_e) tenv1 tenv3 (* ⇒ Γc; ↑γ₁ ⊢〚(v IN e₀ AS b)::↑eᶠ〛ᶠ ⇓ ↑γ₃ *)
    with oql_from_in_sem : string -> oql_expr -> list oql_env -> list oql_env -> Prop :=
    | oql_from_in_sem_nil v e :
        oql_from_in_sem v e nil nil (* Γc; []; v ⊢〚e〛ⁱ ⇓ [] *)
    | oql_from_in_sem_cons v e env tenv1 tenv2 tenv3 dl :
        oql_from_in_sem v e tenv1 tenv2 -> (* Γc; ↑γ₁; v ⊢〚e〛ⁱ ⇓ ↑γ₂ *)
        oql_expr_sem e env (dcoll dl) -> (* ∧ Γc; γ ⊢〚e〛⇓ ↑dl *)
        env_map_concat_single env (map (fun x => ((v,x)::nil)) dl) = tenv3 -> (* ∧ ↑γ₃ = mapc γ v ↑dl *)
        oql_from_in_sem v e (env :: tenv1) (tenv3 ++ tenv2) (* ⇒ Γc; γ::↑γ₁; v ⊢〚e〛ⁱ ⇓ ↑γ₃⊕↑γ₂ *)
    with oql_from_in_cast_sem : string -> list string -> oql_expr -> list oql_env -> list oql_env -> Prop :=
    | oql_from_in_cast_sem_nil v brands e :
        oql_from_in_cast_sem v brands e nil nil (* Γc; []; v; b ⊢〚e〛ᶜ ⇓ [] *)
    | oql_from_in_cast_sem_cons v brands e env tenv1 tenv2 tenv3 dl dl' :
        oql_from_in_cast_sem v brands e tenv1 tenv2 -> (* Γc; ↑γ₁; v; b ⊢〚e〛ᶜ ⇓ ↑γ₂ *)
        oql_expr_sem e env (dcoll dl) -> (* ∧ Γc; γ ⊢〚e〛⇓ ↑dl *)
        filter_cast_sem brands dl dl' -> (* ∧ b₂ ⊢〚↑dl〛ᶜ ⇓ ↑dl' *)
        env_map_concat_single env (map (fun x => ((v,x)::nil)) dl') = tenv3 ->(* ∧ ↑γ₃ = mapc v ↑dl' *)
        oql_from_in_cast_sem v brands e (env :: tenv1) (tenv3 ++ tenv2) (* ⇒ Γc; γ::↑γ₁; v ⊢〚e〛ᶜ ⇓ ↑γ₃⊕↑γ₂ *)
    with oql_where_sem : oql_expr -> list oql_env -> list oql_env -> Prop :=
    | oql_where_sem_nil e :
        oql_where_sem e nil nil (* Γc; [] ⊢〚e〛ʷ ⇓ [] *)
    | oql_where_sem_true e env tenv1 tenv2 :
        oql_where_sem e tenv1 tenv2 -> (* Γc; ↑γ₁ ⊢〚e〛ʷ ⇓ ↑γ₂ *)
        oql_expr_sem e env (dbool true) -> (* ∧ Γc; γ ⊢〚e〛⇓ true *)
        oql_where_sem e (env :: tenv1) (env :: tenv2) (* ⇒ Γc; γ::↑γ₁ ⊢〚e〛ʷ ⇓ γ::↑γ₂ *)
    | oql_where_sem_false e env1 tenv1 tenv2 :
        oql_where_sem e tenv1 tenv2 -> (* Γc; ↑γ₁ ⊢〚e〛ʷ ⇓ ↑γ₂ *)
        oql_expr_sem e env1 (dbool false) -> (* ∧ Γc; γ ⊢〚e〛⇓ false *)
        oql_where_sem e (env1 :: tenv1) tenv2 (* ⇒ Γc; γ::↑γ₁ ⊢〚e〛ʷ ⇓ ↑γ₂ *)
    with oql_order_sem : oql_expr -> SortDesc -> list oql_env -> list oql_env -> Prop :=
    | oql_order_sem_sorting e sc tenv1 tenv2 :
        Permutation tenv1 tenv2 -> (* ⇒ ↑γ₂ permutation of ↑γ₁ *)
        oql_strongly_sorted e sc tenv2 -> (* ⇒ Γc ⊢ ↑γ₂ strongly sorted wrt e+sc *)
        oql_order_sem e sc tenv1 tenv2 (* ⇒ Γc; ↑γ₁ ⊢〚e〛ᵒ ⇓ ↑γ₂ *)
    with oql_select_sem: oql_select_expr -> list oql_env -> data -> Prop :=
    | sem_OSelect e : forall tenv dl,
        oql_select_map_sem e tenv dl -> (* Γc; ↑γ ⊢〚e〛ᵐ ⇓ ↑dl *)
        oql_select_sem (OSelect e) tenv (dcoll dl) (* ⇒ Γc; ↑γ ⊢〚e〛ˢ ⇓ {↑dl} *)
    | sem_OSelectDistinct e : forall tenv dl dl',
        oql_select_map_sem e tenv dl -> (* Γc; ↑γ ⊢〚e〛ᵐ ⇓ ↑dl *)
        bdistinct dl = dl' -> (* ∧ ↑dl' = distinct ↑dl *)
        oql_select_sem (OSelectDistinct e) tenv (dcoll dl') (* ⇒ Γc; ↑γ ⊢〚e〛ˢ ⇓ {↑dl'} *)
    with oql_select_map_sem : oql_expr -> list oql_env -> list data -> Prop :=
    | oql_select_map_sem_nil e :
        oql_select_map_sem e nil nil
    | oql_select_map_sem_cons e env tenv1 d dl1:
        oql_expr_sem e env d -> (* Γc; γ ⊢〚e〛⇓ d *)
        oql_select_map_sem e tenv1 dl1 -> (* ∧ Γc; ↑γ₁ ⊢〚e〛ᵐ ⇓ ↑dl *)
        oql_select_map_sem e (env :: tenv1) (d :: dl1) (* ⇒ Γc; γ::↑γ₁ ⊢〚e〛ᵐ ⇓ d::↑dl *)
    with oql_strongly_sorted : oql_expr -> SortDesc -> list oql_env -> Prop :=
    | oql_SSorted_nil e sc : oql_strongly_sorted e sc nil
    | oql_SSorted_cons e sc env1 d1 sd1 tenv :

        sdata_of_data d1 = Some sd1 ->

      dict_field_le (sdl1, d1) (sd2::nil, d2) ->
      oql_sort_criteria e d1 sdl1 env2
