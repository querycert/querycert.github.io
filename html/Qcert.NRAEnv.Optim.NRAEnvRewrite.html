  Fixpoint nraenv_core_split_last (e:nraenv_core) : (nraenv_core*nraenv_core)
    := match e with
         | ANID => (ANID,ANID)
         | ANConst c => (ANID,ANConst c)

         | ANUnop u e1 =>

         | ANMapConcat e1 e2 =>
