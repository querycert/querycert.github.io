  Definition alg_of_pand (p1 p2:pat) : alg :=
    alg_of_pat (pand p1 p2).

  Definition alg_for_pand (q1 q2:alg) : alg :=
    (♯flatten(χ⟨q2
             ⟩( unnest_two "PBIND1" "PBIND"
                  (χ⟨‵[| ("PDATA", (ID) · "PDATA")|]
                     ⊕ (‵[| ("PCONST", (ID) · "PCONST")|]
                        ⊕ ‵[| ("PBIND1", (ID) · "PBIND" ⊗ (ID) · "PBIND1")|])
                   ⟩( unnest_two "a1" "PBIND1"
                        (‵{|ID
                            ⊕ ‵[| ("a1",
                              χ⟨‵[||] ⟩( σ⟨ID ⟩( q1)))|]|}))))))%alg.

  Lemma alg_of_pand_works (p1 p2:pat) :
    alg_of_pat (pand p1 p2) = alg_for_pand (alg_of_pat p1) (alg_of_pat p2).

  Definition algenv_of_pand (p1 p2:pat) : algenv :=
    algenv_of_pat (pand p1 p2).

  Definition algenv_for_pand (q1 q2: algenv) : algenv :=
    ANUnop AFlatten
         (ANAppEnv (ANMapEnv q2)
            (ANUnop AFlatten
               (ANMap (ANBinop AMergeConcat ANEnv ANID)
                  (ANMap (ANConst (drec nil))
                     (ANSelect ANID q1))))).

  Lemma algenv_of_pand_works (p1 p2:pat) :
    algenv_of_pat (pand p1 p2) = algenv_for_pand (algenv_of_pat p1) (algenv_of_pat p2).
