  Definition nraenv_eq (op1 op2:nraenv) : Prop :=
    ∀ (h:list(string×string))
           (c:list (string×data))
           (dn_c: Forall (fun d : string × data ⇒ data_normalized h (snd d)) c)

           (dn_env:data_normalized h env)
           (x:data)
           (dn_x:data_normalized h x),
