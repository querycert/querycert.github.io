  Lemma ATdot {τc p s τin τ pf τout k}:
    p ▷ τin >=> Rec k τ pf ⊣ τc ->
    tdot τ s = Some τout ->
    AUnop (ADot s) p ▷ τin >=> τout ⊣ τc.
Proof.
    intros.
    repeat econstructor; eauto.
  Qed.

  Lemma ATdot_inv {τc p s τin τout}:
    AUnop (ADot s) p ▷ τin >=> τout ⊣ τc ->
    exists τ pf k,
      p ▷ τin >=> Rec k τ pf ⊣ τc /\
      tdot τ s = Some τout.
Proof.
    inversion 1; subst.
    inversion H2; subst.
    repeat econstructor; eauto.
  Qed.

  Qed.
